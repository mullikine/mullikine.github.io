#!/bin/bash
export TTY

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

term="$@"
term="$(p "$term" | urlencode)"

curl "http://suggestqueries.google.com/complete/search?&output=toolbar&gl=us&hl=en&q=$term%20vs" |
    xml2json |
    jq -r '.toplevel.CompleteSuggestion[].suggestion.data' | sed "s/^$term vs \?//" | uniqnosort | sed "/^$/d" | pavs
1
