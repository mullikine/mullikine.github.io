-- The number we need to get to the next multiple of 5
needfornext5 :: Int -> Int
needfornext5 x = 5 - x `mod` 5

-- m5 = the next multiple of 5
-- | is a condition
round5 :: Int -> Int
round5 x
       | x < 38 = x
       | (m5 - x) < 3 = m5
       | otherwise = x
       where m5 = x + (needfornext5 x)
