(defun lift-to-list (thing)
    (etypecase
        thing
      (string (list thing))
      (list thing)))

;; (print (length (split-lines "")))


(defun lift-to-list-if-multiline (thing)
  (if (and (stringp thing) (< 1 (length (split-lines thing))))
      (split-lines thing)
      thing))


(defun split (chars str &optional (lst nil) (accm ""))
  (cond
    ((= (length str) 0) (reverse (cons accm lst)))
    (t
     (let ((c (char str 0)))
       (if (member c chars)
           (split chars (subseq str 1) (cons accm lst) "")
           (split chars (subseq str 1)
                  lst
                  (concatenate 'string
                               accm
                               (string c))))))))


(defun join (str lst &optional (jstr ""))
  (cond
    ((null lst) jstr)
    (t (let ((news (concatenate 'string
                                jstr
                                (first lst)
                                (if (null (rest lst))
                                    ""
                                    str))))
         (join str (rest lst) news)))))


(defun combinations (first &rest strings)
  (if (null strings)

      (lift-to-list first)

      (loop for first-choice in (if (listp first) first (list first))

            append (loop for comb in (apply #'combinations strings)
                         collect (join " " `(,first-choice ,comb))))))

#!/bin/zsh
export TTY

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

args="$(
for var in "$@"
do
    if test -e "$var"; then
        echo -n " =(cat \"$var\" | q -l | s chomp)"
    else
        echo -n " =(printf -- "%s" \"$var\" | q -l | s chomp)"
    fi
done
)"

eval "combinations.cl$args"
