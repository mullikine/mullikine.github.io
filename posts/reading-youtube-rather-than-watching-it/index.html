#!/bin/bash
export TTY

auto_only=n
while [ $# -gt 0 ]; do opt="$1"; case "$opt" in
    -a) {
        auto_only=y
        shift
    }
    ;;

    *) break;
esac; done

url="$1"
shift

mkdir -p $DUMP$NOTES/ws/youtube/subs
cd $DUMP$NOTES/ws/youtube/subs

clean_while_downloading() {
    # awk 1 does not flush by default
    awk1 | while IFS=$'\n' read -r line; do
        timeout 2 bash -c "until test -s \"$line\"; do sleep 0.5; done"
        # here, need to wait until the file is written to
        # a beep
        if [ -f "$line" ]; then
            cat "$line" | clean-subs | sponge "$line"
            printf -- "%s\n" "$line" | sed "s=^=$(pwd)/="
        fi
    done
}

# We prefer real subs, but will accept auto-generated subs too

if ! test "$auto_only" = "y"; then
    path="$(youtube-dl --sub-lang en --all-subs --skip-download "$url" 2>/dev/null | soak | sed -u -n 's/.*Writing video subtitles to: \(.*\)/\1/p' | tee /dev/stderr | clean_while_downloading)"
    printf -- "%s\n" "$path"
fi

if ! [ -n "$subs" ] && ! test -n "$path"; then
    echo "getting auto subs" 1>&2
    youtube-dl --sub-lang en --write-auto-sub --skip-download "$url" 2>/dev/null | soak | sed -u -n 's/.*Writing video subtitles to: \(.*\)/\1/p' | tee /dev/stderr | clean_while_downloading
fi

#!/bin/bash
export TTY

width="$1"
# : ${width:="38"} # phone
: ${width:="46"}

tf_input="$(ux tf input || echo /dev/null)"
trap "rm \"$tf_input\" 2>/dev/null" 0

cat > "$tf_input"

initalwhitespace="$(cat "$tf_input" | sed -n "1s/^\(\s\+\).*/\1/p")"
whitespacewidth="$(p "$initalwhitespace" | wc -c)"
width="$(( width - whitespacewidth ))"

cat "$tf_input" |
    sed 's/\[[0-9]\]//g' |
    sed 's/^\(\s\+\)//' |
    s join ' ' |
    mnm-text |
    sed 's/ \+/ /g' |
    segment-sentences |
    wrlp -E "wrap $width; echo" |
    sed 's/^\s\+$//' |
    sed "s/\s\+/ /g" |
    sed '${/^$/d;}' |
    sed "s/^/$initalwhitespace/"

#!/bin/bash
export TTY

# Collect history of all readsubs invocations
( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

while [ $# -gt 0 ]; do opt="$1"; case "$opt" in
    "") { shift; }; ;;
    -u) {
        export UPDATE=y
        shift
    }
    ;;

    *) break;
esac; done

last_arg="${@: -1}"
set -- "${@:1:$(($#-1))}" # shift last arg

url="$last_arg"
url="$(printf -- "%s" "$url" | xurls)"

cd "$DUMP$NOTES/ws/youtube/subs"

is_tty() {
    # If stout is a tty
    [[ -t 1 ]]
}

# Cache the subtitles after the first invocation
# can't use xa here
ci -E "yt-subs.sh \"$url\" | grep -i \"\\.en\\.\" | head -n 1 | s chomp | bs ' ' | xargs -l cat | tr -s '\\n' ' ' | sed 's/^\s\+//' | segment-sentences | ttp | tf txt" | {
    if is_tty; then
        xa vs "$@"
    else
        xa cat | pager
    fi
    # pager -v
}

# If readsubs stops working, try this
# sudo pip install --upgrade youtube_dl
1

readsubs "https://www.youtube.com/watch?v=eX9m3g5J-XA"
