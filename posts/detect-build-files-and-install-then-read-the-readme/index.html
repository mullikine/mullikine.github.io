#!/bin/bash
export TTY

unset CWD

# TODO Add something to activate a venv
# x -sh bash -s '. .env/bin/activate' -c m -i

# DONE Add something to run a file *AS* a type of file (such as a Dockerfile)
# vim +/"-spec) {" "$SCRIPTS/cr"

export NOPAK

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

# quickrun.el might be the emacs equivalent of cr

# Example
# $HOME/notes2018/ws/kotlin/examples/ls

# Everything related to compiling and running things
# Zr in vim should link to this

cd_run() {
    CWD="$1" exec x -zsh -s "ls" -c m -s "./$mant" -c m -a
}

cd_ls() {
    CWD="$1" exec x -zsh -s "ls" -c m -a
}

sn="$(basename "$0")"

case "$sn" in
    cri) {
        interactive=y
    }
    ;;

    *)
esac

sudomaybe() {
    if test "$DO_SUDO" = "y"; then
        cmd msudo "$@" 1>&2
        msudo "$@"
    else
        "$@"
    fi
}

while [ $# -gt 0 ]; do opt="$1"; case "$opt" in
    -i) {
        interactive=y
        shift
    }
    ;;

    -y) {
        export YN_DEFAULT=y
        shift
    }
    ;;

    -spec) {
        fnspec="$2"
        shift
        shift
    }
    ;;

    -ft*) { # Because "$1" == "-ft kotlin", when used in shebang
        # echo "$opt"
        if pl "$opt" | grep -q -P -- '-ft.+'; then
            file_type="$(p "$opt" | sed 's/^-ft *//')"
            # file_type="${opt##* }"
            shift
        else
            file_type="$2"
            shift
            shift
        fi
    }
    ;;

    *) break;
esac; done

# echo "$file_type"
# exit 0

# printf -- "%s\n" "$1"

stdin_exists() {
    ! [ -t 0 ]
}

if stdin_exists && ! test -f "$1"; then
    fp="$(cat | tf txt)"
else
    fp="$1"
    shift
fi

if ! test -f "$1" && test -d "$fp"; then
    cd "$fp"
    fp=
fi

trycr() {
    fn="$1"

    if ! test -f "$fn"; then
        possible="$(find . -maxdepth 1 -iname "$fn" -print -quit)"
        if test -n "$possible"; then
            fn="$possible"
        fi
    fi

    if test -f "$fn" && yn "Use $fn?"; then
        cr "$1"
        return 1
    fi

    return 0
}

if test -z "$fp"; then
    # Look for build files

    echo cr
    echo Looking for build files | udl

    # hsq cr | xa cat1 | scrape "\bcr [^ ]+" | uniqnosort | grep -v / | v

    for f in install*; do
        test -x "$f" && trycr "$f"
    done
    trycr build.sh
    trycr docker-compose.yml
    for f in Dockerfile*; do
        trycr "$f"
    done
    trycr appveyor.yml
    trycr CMakeLists.txt # Keep this before Makefile
    trycr Makefile
    trycr project.clj
    trycr composer.json
    trycr Pipfile
    trycr default.nix
    trycr Rakefile
    trycr mkdocks
    trycr meson.build
    trycr build.sbt
    trycr Gemfile
    trycr pom.xml
    trycr psc-package.json
    trycr bower.json
    trycr spago.dhall
    trycr Gruntfile.js
    trycr Project.toml
    trycr autogen.sh
    trycr go.mod
    trycr configure
    trycr main.go
    trycr yarn.lock
    trycr brower.json
    trycr package.json
    trycr package.yaml # hpack
    trycr Setup.hs
    for f in *.cabal; do
        trycr "$f"
    done
    trycr stack.yaml
    trycr pyproject.toml
    trycr setup.py
    trycr configure.ac
    trycr Cargo.toml
    trycr requirements.txt
    trycr DotNetCoreKoans.sln
    trycr Makefile.PL
    trycr dune-project
    trycr config.toml

    glow

    exit 1
fi

rp="$(realpath "$fp")"
dn="${rp%/*}"
