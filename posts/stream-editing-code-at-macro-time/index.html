#!/bin/bash
export TTY

: ${SEDOPTS:="-u"}

. ~/.shell_environment

sn="$(basename "$0")"

# Use arrays here

unminimise() {
    sed $SEDOPTS -e "s#\$DUMP#${DUMP/}#g" \
                 -e "s#\$NOTES#${NOTES}#g" \
                 -e "s#\$BULK#${BULK}#g" \
                 -e "s#\$EMACSD#${EMACSD}#g" \
                 -e "s#\$VIMCONFIG#${VIMCONFIG}#g" \
                 -e "s#\$VIMSNIPPETS#${VIMSNIPPETS}#g" \
                 -e "s#\$HOME#${HOME}#g" \
                 -e "s#\$FSNOTES#${FSNOTES}#g" \
                 -e "s#\${GOPATH%%:\*}#${GOPATH%%:\*}#g" \
                 -e "s#\$MYGIT#${MYGIT}#g" \
                 -e "s#\$SCRIPTS#${SCRIPTS}#g" \
                 -e "s#\$BUILD#${BUILD}#g" \
                 -e "s#\$LOGDIR#${LOGDIR}#g" \
                 -e "s#^~/#${HOME}/#g" \
                 -e "s#^~\$#${HOME}#g" \
                 | cat
}

minimise() {
    sed $SEDOPTS -e "s#/morgan##g" \
                 -e "s#/var/smulliga#${HOME}#g" \
                 -e "s#${HOME}${HOME}#${HOME}#g" \
                 -e "s#${NOTES}2018#\$NOTES#g" \
                 -e "s#${NOTES}#\$NOTES#g" \
                 -e "s#^/home/[a-zA-Z]\+#$HOME#g" \
                 -e "s#\([\"' =]\)/home/[a-zA-Z]\+#\1$HOME#g" \
                 -e "s#^/Users/[^/]\+#\$HOME#g" \
                 -e "s#\([\"' =]\)/Users/[a-zA-Z]\+#\1$HOME#g" \
                 -e "s#${DUMP}#\$DUMP#g" \
                 -e "s#${VIMSNIPPETS}#\$VIMSNIPPETS#g" \
                 -e "s#${VIMCONFIG}#\$VIMCONFIG#g" \
                 -e "s#${LOGDIR}#\$LOGDIR#g" \
                 -e "s#${BULK}#\$BULK#g" \
                 -e "s#${EMACSD}#\$EMACSD#g" \
                 -e "s#${SCRIPTS}#\$SCRIPTS#g" \
                 -e "s#${MYGIT}#\$MYGIT#g" \
                 -e "s#${GOPATH%%:*}#\${GOPATH%%:\*}#g" \
                 -e "s#${BUILD}#\$BUILD#g" \
                 -e "s#${FSNOTES}#\$FSNOTES#g" \
                 -e "s#${HOME}#\$HOME#g" \
                 -e "s#\$HOME/source/git#\$MYGIT#g" \
                 | cat
}

if test "$1" = "-x" || test "$sn" = "umn" || test "$sn" = "unminimise.sh"; then
    unminimise
elif test "$sn" = "m" || test "$sn" = "mnm" || test "$sn" = "minimise.sh"; then
    unminimise | minimise
else
    cat
fi

(defmacro macro-unminimise (&rest body)
  "This unminimises the code"
  (let* ((codestring (pp body))
         (ucodestring (umn codestring))
         (newcode (my-eval-string (concat "'" ucodestring))))
    `(progn ,@newcode)))
(defalias 'mu 'macro-unminimise)

(defmacro macro-minimise (&rest body)
  "This minimises the code"
  (let* ((codestring (pp body))
         (mcodestring (mnm codestring))
         (newcode (my-eval-string (concat "'" mcodestring))))
    `(progn ,@newcode)))
(defalias 'mm 'macro-minimise)

(defun my-eval-string (string)
  "Evaluate elisp code stored in a string."
  (eval (car (read-from-string (format "(progn %s)" string)))))

(defun pp-oneline (l)
  (chomp (replace-regexp-in-string "\n +" " " (pp l))))
(defalias 'pp-ol 'pp-oneline)

(defun pp-map-line (l)
  (string-join (mapcar 'pp-oneline l) "\n"))

(defmacro macro-sed (expr &rest body)
  "This transforms the code with a sed expression"
  ;; TODO Make this one-linerise each element of body
  (let* ((codestring (pp-map-line body))
         (ucodestring (sed expr codestring))
         (newcode (my-eval-string (concat "'(progn " ucodestring ")"))))
    ;; `(progn ,@newcode)
    newcode))
(defalias 'ms 'macro-sed)
