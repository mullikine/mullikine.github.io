#!/bin/bash
export TTY

EXACT=y
while [ $# -gt 0 ]; do opt="$1"; case "$opt" in
    "") { shift; }; ;;
    -x) {
        EXACT=y
        shift
    }
    ;;

    -X) {
        EXACT=n
        shift
    }
    ;;

    *) break;
esac; done

# Remove all options
# This ruins the command-line invocation -- fix this
# eval "set -- $(cmd "$@" | sed 's/\(^\| \)-[^ ]\+\( \|$\)/ /g')"
eval "set -- $(cmd "$@" | sed-pcre "s/^((^| )-[^ ]+)+//")"
# Remove all flags up until the first non-flag
# It should not be allowed to strip -->

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

query="$@"

is_tty() {
    # If stout is a tty
    [[ -t 1 ]]
}

# Keep in mind that permutations of arguments is often desired

# Going to the website works for
# stack exec -- hoogle "--count=10000" "[String] -> String" | v
# stack exec -- hoogle "--count=10000" "$(pl "[String] -> String" | urlencode)" | v
# https://hoogle.haskell.org/?hoogle=%5BString%5D+-%3E+String&scope=set%3Astackage


# The grep -F is for ensuring I get the results I expect
# hs-type-declarative-search "[String] -> String"

grepit () {
    if test "$EXACT" = "y"; then
        grep -P -- ":: $query$"
    else
        grep -F -- "$query"
    fi
}

# echo "$query" 1>&2
# exit 0
stack exec -- hoogle --count=10000 "$query" | grepit | {
    if is_tty; then
        vs
    else
        cat
    fi
}

#!/bin/bash
export TTY

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

type="$1"

hs-type-declarative-search "$type" |
    fzf -p -pscript hs-type-declarative-search-preview |
    sed -e 's/ /./' -e "s/ ::.*//" | pager -vs
