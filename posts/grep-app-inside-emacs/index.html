grep.app inside emacs

Aug 23, 2020
4 min read
iremacselisp

Summary

grep.app is a free web app that lets you perform full-text search on GitHub code, using regex to match content, file paths and repository names.

The problem

I’d like to make this more convenient by searching and obtaining example code all within emacs.

  • emacs’ eww-mode doesn’t support javascript.
  • I’d like to present the website in a usable form – code snippets cleaned up.
  • I’d like to automate the process of searching so that it doesn’t require any typing.

(defun grep-app (pattern &optional path repo)
  (interactive (if (selected)
                   (list (selection)
                         (let ((path (get-path)))
                           (if path
                               (file-name-extension path)
                             (read-string-hist "grep.app path:")))
                         nil)
                 (list (read-string-hist "grep.app pattern:")
                       (read-string-hist "grep.app path:")
                       (read-string-hist "grep.app repo:"))))

  (let ((do-literal (selected)))

    (if (and (selected) (s-matches-p "^[a-z0-9]+$" pattern))
        (progn
          (setq do-literal nil)
          (setq pattern (concat "\\b" pattern "\\b"))))

    (setq pattern (urlencode pattern))
    (setq path (urlencode path))
    (setq repo (urlencode repo))

    (let ((url "https://grep.app/search?q="))
      (if (not (string-empty-p pattern))
          (if do-literal
              (setq url (concat url pattern "&case=true"))
            (setq url (concat url pattern "&regexp=true&case=true"))))
      (if (not (string-empty-p path)) (setq url (concat url "&filter[path.pattern][0]=" path)))
      (browse-url-generic url))))

(defun eww-display-html (charset url &optional document point buffer encode)
  (unless (fboundp 'libxml-parse-html-region)
    (error "This function requires Emacs to be compiled with libxml2"))
  (unless (buffer-live-p buffer)
    (error "Buffer %s doesn't exist" buffer))
  (setq url-queue nil)
  (let* ((html (if eww-use-chrome
                   ;; (sn (concat "dom-dump  " (q url)))
                   (if eww-update-ci
                       (shell-command-to-string (concat "UPDATE=y dom-dump " (shell-quote-argument url)))
                     (shell-command-to-string (concat "dom-dump " (shell-quote-argument url))))
                 ""))
         (document
          (or nil ;; document
              (list
               'base (list (cons 'href url))
               (progn
                 (save-excursion
                   ;; Go to after the http headers
                   (progn
                     (beginning-of-buffer)
                     (search-forward-regexp "^$")
                     (search-forward-regexp "^.")
                     (beginning-of-line))
                   ;; Replace the html with the DOM dump
                   (if eww-use-chrome
                       (save-mark-and-excursion
                         ;; Delete from here to the end. Replace with the new html
                         (kill-region (point) (point-max))
                         (insert (encode-coding-string html 'utf-8)))))
                 (setq encode (or encode charset 'utf-8))
                 (condition-case nil
                     (decode-coding-region (point) (point-max) encode)
                   (coding-system-error nil))
                 (save-excursion
                   ;; Remove CRLF before parsing.
                   (while (re-search-forward "\r$" nil t)
                     (replace-match "" t t)))
                 (libxml-parse-html-region (point) (point-max))))))
         (source (and (null document)
                      (buffer-substring (point) (point-max)))))
    (with-current-buffer buffer
      (setq bidi-paragraph-direction nil)
      ;; (plist-put eww-data :source source)
      ;; (new-buffer-from-string (sn (concat "dom-dump " (q "http://google.com"))))
      (plist-put eww-data :source html)
      ;; (plist-put eww-data :source (tv source))
      (plist-put eww-data :dom document)
      (let ((inhibit-read-only t)
            (inhibit-modification-hooks t)
            (shr-target-id (url-target (url-generic-parse-url url)))
            (shr-external-rendering-functions
             (append
              shr-external-rendering-functions
              '((title . eww-tag-title)
                (form . eww-tag-form)
                (input . eww-tag-input)
                (button . eww-form-submit)
                (textarea . eww-tag-textarea)
                (select . eww-tag-select)
                (link . eww-tag-link)
                (meta . eww-tag-meta)
                (a . eww-tag-a)))))
        (erase-buffer)
        (shr-insert-document document)
        (cond
         (point
          (goto-char point))
         (shr-target-id
          (goto-char (point-min))
          (let ((point (next-single-property-change
                        (point-min) 'shr-target-id)))
            (when point
              (goto-char point))))
         (t
          (goto-char (point-min))
          ;; Don't leave point inside forms, because the normal eww
          ;; commands aren't available there.
          (while (and (not (eobp))
                      (get-text-property (point) 'eww-form))
            (forward-line 1)))))
      (eww-size-text-inputs))))

(defun clean-up-grep-app ()
  (interactive)
  (beginning-of-buffer)
  (toggle-read-only)
  (cua-set-mark)
  (search-forward-regexp "^Showing")
  (search-forward-regexp "^\*")
  (my/c-w-cut)
  (toggle-read-only))


(defvar eww-patchup-url-alist nil)
(add-to-list 'eww-patchup-url-alist '("://grep.app/" . clean-up-grep-app))

(defun finished-loading-page ()
 "Needed so that x/expect can automate it"

  ;; Clean up some pages
  (ignore-errors
    (let ((f (assoc-default (get-path) eww-patchup-url-alist #'string-match)))
      (if f (call-function f))))

  ;; enable this by default
  ;; (eww-readable)
  (rename-eww-buffer-unique)
  (message (concat "Finished loading page! [chrome: " (if eww-use-chrome "on" "off") ", cache: " (if (s-matches-p "webcache.google" (get-path)) "on" "off")
                   ", update: " (if eww-update-ci "on" "off") "]"))
  (recenter-top))

(setq eww-after-render-hook '())
(add-hook 'eww-after-render-hook #'finished-loading-page)
