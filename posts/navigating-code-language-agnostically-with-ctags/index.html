(defun cq-classes ()
  "list classes"
  (sh-notty "quiet cq-classes" nil (replace-regexp-in-string "/$" "" (e/chomp (e/pwd)))))

(defun cq-symbols ()
  "list symbols"
  (sh-notty "quiet cq-symbols" nil (replace-regexp-in-string "/$" "" (e/chomp (e/pwd)))))

(defun cq-functions ()
  "list symbols"
  (sh-notty "quiet cq-functions" nil (replace-regexp-in-string "/$" "" (e/chomp (e/pwd)))))

(defun fz-cq-classes ()
  (interactive)
  (goto-cq-sel (fz (b-tabulate (cq-classes)))))

(defun fz-cq-symbols ()
  (interactive)
  (goto-cq-sel (fz (b-tabulate (cq-symbols)))))

(defun fz-cq-functions ()
  (interactive)
  (goto-cq-sel (fz (b-tabulate (cq-functions)))))

(defun goto-cq-sel (sel)
  (interactive)
  (if (not (empty-string-p sel))
      (let* ((target (e/chomp (bp s field 2 sel)))
             (path (e/chomp (bp cut -d : -f 1 target)))
             (line (e/chomp (bp cut -d : -f 2 target)))
             (toplevel (e/chomp (b vc get-top-level))))
        (with-current-buffer
            (find-file (concat toplevel "/" path))
          (goto-line (string-to-int line))))))

(defun show-ctags-for-buffer ()
  (interactive)
  ;; (nw (concat "ctags-hr " (q buffer-file-name) " | sp"))
  (new-buffer-from-string (sn (concat "ctags-hr " (q buffer-file-name)))))

#!/bin/bash

  exec 1>&2

is-git && cd "$(vc get-top-level)"


CWD="$(pwd)"
builtin cd /

dirslug="$(p "$CWD" | sed 's=/$==' | slugify)"
infodir=$HOME/notes/programs/cscope-ctags
destdir="$infodir/$dirslug"

while [ $# -gt 0 ]; do opt="$1"; case "$opt" in
    "") { shift; }; ;;
    -cd) {
        zcd "$destdir"
        exit $?
        shift
    }
    ;;

    *) break;
esac; done

: ${REBUILD:="y"}

if test "$REBUILD" = "y"; then
    # rm -f cscope.{files,*.out}
    rm -rf "$destdir"
fi

mkdir -p "$destdir"

find -L -O3 $CWD \
    -true \
    -and \
    \( \
    -name '*.py' \
    -o -name '*.php' \
    -o -name '*.java' \
    -o -iname '*.[CH]' \
    -o -name '*.cpp' \
    -o -name '*.go' \
    -o -name '*.cs' \
    -o -name '*.hpp'  \
    -o -name '*.cc' \
    -o -name '*.rs' \
    -o -name '*.hh' \
    -o -name '*.hs' \
    -o -name '*.js' \
    -o -name '*.ts' \
    \) \
    > "$destdir/cscope.files"
#    2>&1 > $CWD/cscope.files | cut -c-$COLUMNS
export CSCOPE_DB=$destdir/cscope.out
builtin cd "$destdir"
# -c: needed for codequery
# -b: just build
# -q: create inverted index


if ( cd "$CWD"; show-extensions-below ) | grep -q ',py'; then
    pycscope -i  "$destdir/cscope.files"
else
    cscope -c -b -q
fi

$HOME/scripts/ctags --fields=+latinK+i -R -L ./cscope.files

$MYGIT/Guad/tags2uml/tags2uml --members=1 --methods=1 --infile tags | tee uml.dot | dot-qdot | tv &>/dev/null

set -m
cqmakedb -s ./codequery.db -c ./cscope.out -t ./tags -p

#!/bin/bash
export TTY

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

#    1: Symbol (default)
#    2: Function or macro definition
#    3: Class or struct
#    4: Files including this file
#    5: Full file path
#    6: Functions calling this function
#    7: Functions called by this function
#    8: Calls of this function or macro
#    9: Members and methods of this class
#   10: Class which owns this member or method
#   11: Children of this class (inheritance)
#   12: Parent of this class (inheritance)
#   13: Functions or macros inside this file

sn="$(basename "$0")"

while [ $# -gt 0 ]; do opt="$1"; case "$opt" in
    "") { shift; }; ;;
    -u) {
        UPDATE=y
        shift
    }
    ;;

    *) break;
esac; done

num="1"
case "$sn" in
    cq-functions) {
        num=6
    }
    ;;

    cq-classes) {
        num=3
    }
    ;;

    cq-symbols) {
        num=1
    }
    ;;

    *)
esac

#if ! test -f codequery.db; then
#    cscope_gen.sh
#fi

query="$@"
: ${query:="*"}

is-git && cd "$(vc get-top-level)"
CWD="$(pwd)"

dirslug="$(p "$CWD" | sed 's=/$==' | slugify)"
infodir=$HOME/notes/programs/cscope-ctags
destdir="$infodir/$dirslug"

if test "$UPDATE" = y || ! test -f "$destdir/codequery.db"; then
    cscope_gen.sh
fi

cqsearch -s "$destdir/codequery.db" -p "$num" -u -t "$query" | sed "s=\t$CWD/=\t="
