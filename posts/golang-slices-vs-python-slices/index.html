Golang slices vs Python slices

Nov 1, 2019
5 min read
golangpython

Golang slices are very limited.

Apart from its syntactic limitations, golang slice syntax also lacks in generics due to only being able to operate on golang slices.

Python, with its duck-typing is able to use slices on anything that’s a collection.

package main

import "fmt"

func main() {
      intSlice := []int{1, 2, 3, 4, 5}
      fmt.Printf("Slice: %v\n", intSlice)

      last := intSlice[len(intSlice)-1]
      fmt.Printf("Last element: %v\n", last)

      // Slices are inclusive:exclusive and unspecified LHS is 0
      // This is 0 exclude 0
      first := intSlice[:0]
      fmt.Printf("First element: %d\n", first)

      second := intSlice[0:1]
      fmt.Printf("Second element: %d\n", second)

      third := intSlice[1:2]
      fmt.Printf("Third element: %d\n", third)

      remove := intSlice[:len(intSlice)-1]
      fmt.Printf("Remove Last: %v\n", remove)
}

#!/bin/bash
export TTY

delim="$1"
: ${delim:="/"}
slicenotation="$2"

# reverses
# : ${slicenotation:="::-1"}

# does nothing
: ${slicenotation:=":"}

tf_go="$(ux tf slice go || echo /dev/null)"
trap "rm \"$tf_go\" 2>/dev/null" 0

go_apppend_line=
if printf -- "%s\n" "$slicenotation" | grep -q -P ':'; then
    go_apppend_line="sl = append(sl, result...)"
else
    go_apppend_line="sl = append(sl, result)"
fi

cat > "$tf_go" <<HEREDOC
package main

import "os"
import "fmt"
import "strings"
import "bufio"
import "io/ioutil"

func main() {
    bytes, _ := ioutil.ReadAll(os.Stdin)

    scanner := bufio.NewScanner(strings.NewReader(string(bytes)))
    for scanner.Scan() {
        var sl []string

        result := strings.Split(scanner.Text(), "$delim")[$slicenotation]

        ${go_apppend_line}

        fmt.Println(strings.Join(sl, "$delim"))
    }
}
HEREDOC

go run "$tf_go" 2>/dev/null

#!/bin/bash
export TTY

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

# This uses "slice notation"

# Split line, get slicenotation of words using python syntax from getopts, then join
# I should generate the python with bash

delim="$1"
: ${delim:="/"}
slicenotation="$2"

# reverses
# : ${slicenotation:="::-1"}

# does nothing
: ${slicenotation:=":"}

read -r -d '' PYCODE <<HEREDOC
import sys

try:
    for line in sys.stdin:
        sub=line.rstrip().split('$delim')[$slicenotation]
        sub = sub if type(sub) == list else [sub]
        sys.stdout.write('$delim'.join(sub) + "\n")
except:
    pass
HEREDOC

python3 -c "$PYCODE"
