Creating a transient-mode for advanced Google search operators, and GitHub too

Nov 7, 2020
5 min read
emacselispgoogletoolingirgithub

Synopsis

I want to be able to search Google easily using advanced operators, but not have to remember, nor type in, the syntax.

I create a user interface to generate the google search query.

Demonstration

Looking for a PDF about the perigee moon, but without the word apogee

Searching for PDFs about transducers that are not to do with clojure

Code

All it does is, given a list of google operaters, it constructs a little gui, which shells out to a script.

The script interprets the arguments with the standard getopt syntax, validates them, and then generates a google query.

The google query is sent back to emacs which runs a web browser on it.

I dont usually make use of the google search operators but i could see myself using this all the time, because its so easy.

#!/bin/bash
export TTY

long_opts="$(cmd -l ext:,filetype:,intext:,intitle:,inurl:,site:,not-ext:,not-filetype:,not-intext:,not-intitle:,not-inurl:,not-site:)"

CMD="$(cmd "$@")"
: ${CMD:="$(cmd "$@")"}

options=$(eval "getopt -o '' $long_opts -- $CMD")
if [ $? != 0 ] ; then echo "Failed parsing options." >&2 ; exit 1 ; fi

eval set -- "$options"

{
while true; do
  case "$1" in
    --not[a-z-]*) {
        key="${1##--not-}"
        val="$2"
        printf -- "%s\n" "-$key:$val"
        shift 2
    } ;;
    --[a-z-]*) {
        key="${1##--}"
        val="$2"
        printf -- "%s\n" "$key:$val"
        shift 2
    } ;;
    -- ) shift; break ;;
    * ) break ;;
  esac
done
for var in "$@"
do
    printf -- "%s\n" "$var"
done
} | q -l | tr '\n' ' '

(require 'transient)
(require 'my-strings)

(defset google-key-value-predicates
  ;; prefix with - to invert i.e. -inurl:
  ;; ext is the same as filetype
  (list ;; "ext"
   "filetype"
   "intext"
   "intitle"
   "inurl"
   "site"))

(defun google-transient-search (&optional args)
  (interactive
   (list (transient-args 'google-transient)))
  (let ((query (cl-sn (concat "gen-google-query " (mapconcat 'q args " ")) :chomp t)))
    (eegr query)))

(defun google-transient-search-with-keywords (&optional args)
  (interactive
   (list (transient-args 'google-transient)))
  (let* ((keywords (read-string-hist "gt keywords:"))
         (query (cl-sn (concat "gen-google-query " (mapconcat 'q args " ") " " keywords) :chomp t)))
    (eegr query)))

(defmacro create-google-transient ()
  (let ((args
         (vconcat (list "Arguments")
                  (append
                   (let ((c 0))
                     (cl-loop for p in google-key-value-predicates do (setq c (1+ c))
                              collect
                              (list
                               (str c) p (concat "--" p "="))
                              collect
                              (list ;; (concat "-" (str c))
                               (concat "-" (str c)) (concat "not" p) (concat "--not-" p "="))))
                   )))
        (actions
         (vconcat (list "Actions"
                        (list "g" "Search g" 'google-transient-search)
                        (list "k" "Search with keywords" 'google-transient-search-with-keywords)))))
    `(define-transient-command google-transient ()
       "Google transient"
       ,args
       ,actions)))

(create-google-transient)

(defalias 'list2vector 'vconcat)

(defun create-my-transient (name kvps searchfun kwsearchfun &optional keywordonly)
  (let ((sym (str2sym (concat name "-transient")))
        (args
         (vconcat (list "Arguments")
                  (append
                   (let ((c 0))
                     (cl-loop for p in kvps do (setq c (1+ c)) collect (list (str c) p (concat "--" p "="))))
                   (let ((c 0))
                     (cl-loop for p in kvps do (setq c (1+ c)) collect (list (concat "-" (str c)) (concat "not" p) (concat "--not-" p "="))))
                   )))
        (actions
         (vconcat
          (if keywordonly
              (list "Actions"
                    (list "k" "Search with keywords" kwsearchfun))
            (list "Actions"
                    (list "s" "Search" searchfun)
                    (list "k" "Search with keywords" kwsearchfun))))))
    (eval `(define-transient-command ,sym ()
             ,(concat (s-capitalize name) " transient")
             ,args
             ,actions))))
