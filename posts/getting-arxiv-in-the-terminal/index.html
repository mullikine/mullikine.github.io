We create a bash script

We name it arxiv-summary.

Cant name this ‘as’ because it interferes with pip.

Sometimes packages installed wtih pip need the assembler.

#!/bin/bash
export TTY
# Collect history of all arxiv-summary invocations
( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

last_arg="${@: -1}"
url="$last_arg"
set -- "${@:1:$(($#-1))}" # shift last arg

# if pl "$url" | grep -q -P '^([a-z-]+/)?[0-9v.]+$'; then
if tt "$url" ~ '^([a-z-]+/)?[0-9v.]+$'; then
    url="https://arxiv.org/abs/$url"
fi
url="$(printf -- "%s" "$url" | xurls)"

cifpname="$(printf -- "%s" "$url" | slugify)"

# Use spacy to post-process the abstract
postprocess() {
    sed -n '/Abstract:/,/^$/{s/^\s\+//;p}' |
        sed -z -e 's/\n/ /g' -e 's/^Abstract: //' -e 's/\s\+$//' |
        strip-latex |
        text-to-paras

    return 0
}

sfp="$0"; declare -fx $(bash -c "source $sfp &> /dev/null; compgen -A function") # export all functions

# sleep 1

# Cache the result
ci -fp "$HOME/notes/ws/arxiv/summaries/${cifpname}.txt" -E "elinks-dump \"$url\" | postprocess" | pager -vs "$@"

# elinks-dump "$url" | sed 's/Abstract:/&\n\n/' | text-to-paras

Scrape emacs for the paper reference, scrape the abstract from the web and present in emacs

(defun get-arxiv-summary-line (&optional line)
  (interactive)

  (if (not line)
      (setq line (str (thing-at-point 'line))))
  (if (not line)
      (error "no line under cursor"))

  (new-buffer-from-string (bp sed -u -n "s=.*: \\(\\([a-z-]\\+/\\)\\?[0-9.v]\\+\\).*=\\1=p" | xa arxiv-summary line)))
