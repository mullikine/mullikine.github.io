#!/bin/bash
export TTY

( hs "$(basename "$0")" "$@" "#" "<==" "$(ps -o comm= $PPID)" 0</dev/null ) &>/dev/null

while [ $# -gt 0 ]; do opt="$1"; case "$opt" in
    "") { shift; }; ;;
    -i) {
        use_module_name=y
        shift
    }
    ;;

    -n) {
        nodownload=y
        shift
    }
    ;;

    -u) {
        UPDATE=y
        shift
    }
    ;;

    *) break;
esac; done

downloadornot() {
    if test "$nodownload" = "y"; then
        pager -v
    else
        sed 1q | xa o
    fi
}

export UPDATE

if test "$use_module_name" = "y"; then
    modulename="$1"
    gr hackage import $modulename | grep hackage.haskell.org/package | downloadornot
elif test "$#" = 2 && test "$1" = import; then
    gr hackage import $2 | grep hackage.haskell.org/package | downloadornot
else
    package="$1"

    echo "http://hackage.haskell.org/package/$package" | downloadornot
fi
