(defun chomp (str)
  "Chomp leading and tailing whitespace from STR for each line."
  (while (string-match "\\`\n+\\|^\\s-+\\|\\s-+$\\|\n+\\'"
                       str)
    (setq str (replace-match "" t t str)))
  str)

(defalias 'q 'shell-quote-argument)

(defun rpl-at-line-p (rpl)
  (let* ((output (chomp (shell-command-to-string (concat "rosie grep -o subs " (q rpl)) (thing-at-point 'line t))))
         (matches (str2list output))
         (n (length matches)))
    (and (not (string-empty-p output)) (> n 0))))

(defun rpl-at-point-p (rpl)
  (let* ((output (chomp (sn (concat "rosie grep -o subs " (q rpl)) (thing-at-point 'sexp t))))
         (matches (str2list output))
         (n (length matches)))
    (and (not (string-empty-p output)) (> n 0))))

Create the plugin

context-tuples is a list of tuples where the first element of each tuple is a list of predicates and the second element is a list of suggested functions.

I use memoisation to ensure that each predicate only evaluates once; That way, it’s possible to reuse predicates without having to evaluate them multiple times.

(defun flatten-once (list-of-lists)
  (apply #'append list-of-lists))

(defalias 'flatten-recursively '-flatten)


(defun hash-expression (expr)
  (sha1 (str expr)))

(defun func-for-expression (expr &optional update)
  (let* ((funcsym (str2sym (concat "my-context-predicate-" (hash-expression expr)))))
    (if (and (not update) (my/function-exists funcsym))
        funcsym
        (eval `(defun ,funcsym ()
                 ,expr)))))

;; This is the list of tuples
(setq context-tuples `((((major-mode-p 'emacs-lisp-mode)
                         (rpl-at-point-p "net.ipv4"))
                        (copy-ip-here))
                       (((major-mode-p 'org-mode)
                         (rpl-at-line-p "net.email"))
                        (copy-email-here))))

(setq context-preds '())
(setq context-pred-funcs '())
(setq context-tuples-compiled '())

(defun compile-context-tuple (context-tuple)
  (let ((pred-funcs (mapcar 'func-for-expression (car context-tuple))))
    (cons pred-funcs (second context-tuple))))

(defun build-context-functions ()
  (interactive)
  (setq context-preds (-distinct (flatten-once (cl-loop for tup in context-tuples collect (car tup)))))

  ;; Go through and make functions first -- a little unnecessary
  (setq context-pred-funcs (cl-loop for pred in context-preds collect (func-for-expression pred)))

  (setq context-tuples-compiled (cl-loop for tup in context-tuples collect (compile-context-tuple tup)))

  )
(build-context-functions)

(defun suggest-funcs ()
  (interactive)
  (cl-loop for f in context-pred-funcs do (ignore-errors (memoize f)))

  (let ((suggestions (cl-loop for tup in context-tuples-compiled collect (if (eval `(and ,@(mapcar 'list (car tup))))
                                                                             (second tup)
                                                                           '()))))

    (cl-loop for f in context-pred-funcs do (ignore-errors (memoize-restore f)))
    (let* ((fz-input (remove nil (-distinct suggestions)))
           (sel (if fz-input
                    (fz fz-input))))
      (if sel
          (let ((selsym (str2sym sel)))
            (if (and (function-p selsym) (commandp selsym))
                (call-interactively selsym)
              (call-function sel)))))))

(define-key global-map (kbd "<help> G") 'suggest-funcs)

(provide 'my-context-functions)
